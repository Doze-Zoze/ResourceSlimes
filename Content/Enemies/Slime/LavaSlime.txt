using Terraria.ModLoader;
using Terraria;
using Terraria.ID;
using Microsoft.Xna.Framework;
using Terraria.ModLoader.Utilities;
using Terraria.GameContent.ItemDropRules;
using Terraria.GameContent.Bestiary;

namespace ResourceSlimes.Enemies
{
    public class LavaSlime : GlobalNPC
    {
        public override void ModifyNPCLoot(NPCLoot npcLoot)
        {
            if (npc.type == NPCID.LavaSlime && NPC.downedBoss2)
            {
                npcLoot.Add(ItemDropRule.Common(ItemID.Hellstone).Type, 4, 1, 10);
                npcLoot.Add(ItemDropRule.Common(ItemID.Obsidian).Type, 8, 1, 10);
            }
            if (npc.type == NPCID.SandSlime && NPC.downedBoss1)
            {
                npcLoot.Add(ItemDropRule.Common(ItemID.DesertFossil).Type, 4, 30, 60);
            }
        }


    }
}
